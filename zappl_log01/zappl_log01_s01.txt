CLASS lcl_event_receiver  DEFINITION .  "定义类 捕捉各种事件
  PUBLIC  SECTION .
    METHODS handle_modify                    "数据改变
    FOR  EVENT data_changed  OF cl_gui_alv_grid

    METHODS handle_data_changed_finished FOR EVENT data_changed_finished OF cl_gui_alv_grid
    IMPORTING e_modified et_good_cells sender."这个方法就会触发事件

FORM 1420_alv_user_command USING r_ucomm LIKE sy-ucomm
      rs_selfield TYPE slis_selfield. "响应ALV点击自定义按钮之后的事件

  DATA: lob_grid TYPE REF TO cl_gui_alv_grid,
        lc_i     TYPE i,
        lv_subrc TYPE sy-subrc.

  CALL FUNCTION 'GET_GLOBALS_FROM_SLVC_FULLSCR'
    IMPORTING
      e_grid = lob_grid.

  CALL METHOD lob_grid->check_changed_data.
  rs_selfield-refresh = 'X'.
  rs_selfield-col_stable = 'X'.
  rs_selfield-row_stable = 'X'.

  CASE r_ucomm.
    WHEN '&IC1'.
    WHEN '&F03' OR '&F15'.
      LEAVE PROGRAM .
    WHEN '&F12'.
      LEAVE PROGRAM.
    WHEN '&DATA_SAVE'.
*应用日志
      READ TABLE gt_alv INTO gs_alv WITH KEY sel = 'X'.
      IF sy-subrc = 0.
        PERFORM 1421_frm_appl_log.
      ELSE.
        MESSAGE '请选择已修改的行!' TYPE 'E'.
      ENDIF.
    WHEN OTHERS.
  ENDCASE.

  rs_selfield-refresh = 'X'.
  CALL METHOD lob_grid->refresh_table_display.



ENDFORM. "ALV_USER_COMMAND

FORM 1430_alv_pf_event USING p_grid TYPE slis_data_caller_exit.

  DATA: r_grid  TYPE REF TO cl_gui_alv_grid.

  CALL FUNCTION 'GET_GLOBALS_FROM_SLVC_FULLSCR'
    IMPORTING
      e_grid = r_grid.

  CALL METHOD r_grid->set_user_command "注册事件
  EXPORTING
    i_ucomm = cl_gui_alv_grid=>mc_fc_data_save. "按钮事件

  CREATE OBJECT gt_event_receiver .
  SET  HANDLER gt_event_receiver->handle_modify  FOR r_grid .
  SET  HANDLER gt_event_receiver->handle_data_changed_finished  FOR r_grid .

  gs_drop-handle = '1'.
  gs_drop-value  = 'A'.
  APPEND gs_drop TO gt_drop.
  CLEAR gs_drop.
  gs_drop-handle = '1'.
  gs_drop-value  = 'B'.
  APPEND gs_drop TO gt_drop.
  gs_drop-handle = '1'.
  gs_drop-value  = 'C'.
  APPEND gs_drop TO gt_drop.
  CLEAR gs_drop.

  CALL METHOD r_grid->set_drop_down_table
    EXPORTING
      it_drop_down = gt_drop.

ENDFORM.                    " ALV_PF_EVENT
