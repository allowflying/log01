FORM frm_get_data .

  CLEAR gt_alv .
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_alv
    FROM zcdslog
   WHERE ztcode IN s_ztcode
     AND zuname IN s_zuname
     AND zdatum IN s_zdatum .

  SORT gt_alv BY zdatum ztimlo DESCENDING .

ENDFORM.                    "frm_get_data
